📋 API 调用方式（点击展开） 
# 通过 curl 升级
curl -X POST http://HOST:5012/api/asap/upgrade/upload \
  -F "file=@upgrade_v1.1.0.zip" \
  -F "remark=修复了xxx问题，新增了xxx功能"

# 升级包内放入 version.json 可自动读取版本信息
# version.json 格式：
# {
#   "title": "v1.1.0 安全更新",
#   "changes": ["修复: xxx", "新增: xxx"]
# }

# 查看升级记录
curl http://HOST:5012/api/asap/upgrade/records

# 回滚到指定备份
curl -X POST http://HOST:5012/api/asap/upgrade/rollback/upgrade_20260101_120000

# 查看当前版本
curl http://HOST:5012/api/asap/upgrade/version
